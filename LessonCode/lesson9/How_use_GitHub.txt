# ===============================
# 📝 Git & GitHub 常用命令速查表（SSH 版）
# ===============================

# 一、配置用户信息（只需一次）
git config --global user.name "你的GitHub用户名"
git config --global user.email "你的GitHub邮箱"

# 二、获取代码（Clone）
git clone git@github.com:用户名/仓库名.git
# 示例：
git clone git@github.com:Jason-zrh/MyLinuxCode.git

# 三、仓库操作
cd MyLinuxCode               # 进入仓库目录
git remote -v                # 查看远程仓库

# 四、提交修改
git status                   # 查看修改
git add 文件名               # 添加单个文件
git add .                    # 添加所有修改
git commit -m "描述本次修改"  # 提交修改

# 五、推送到远程
git push origin main         # 推送到 main 分支
git push -u origin main      # 第一次推送时用，后面只需要 git push

# 六、同步远程更新
git pull origin main         # 拉取远程最新代码

# 七、分支操作
git branch                   # 查看本地分支
git checkout -b 新分支名     # 创建并切换到新分支
git checkout 分支名          # 切换分支
git push origin 新分支名     # 推送新分支到远程

# 八、撤销与回退
git checkout -- 文件名       # 撤销工作区修改（未 add 前）
git reset HEAD 文件名        # 撤销暂存区修改（已 add，未 commit）
git reset --hard HEAD~1      # 回退到上一个版本
